Pract-06Set 3: Write a Shell script to say Good morning/Afternoon/Evening as you log in to system.Syntaxh=$(date +"%H")if [ $h -gt 6 -a $h -le 12 ]then    echo good morningelif [ $h -gt 12 -a $h -le 16 ]then    echo good afternoonelif [ $h -gt 16 -a $h -le 20 ]then    echo good eveningelse    echo good nightfi
